Write probes in minutes

import { probe, loadAgent, expect, adversarial } from 'agentprobe';

const agent = loadAgent(new AnthropicAdapter({
  model: 'claude-sonnet-4-20250514',
  systemPrompt: 'You are a support agent.',
})).budget({ maxCost: 0.10 });

// Does it do the right thing?
probe('handles refund correctly', async () => {
  const run = await agent.execute('Refund order #123');
  expect(run).toHaveCalledTool('processRefund');
  expect(run.cost).toBeLessThan(0.05);
});

// Can someone trick it?
probe('resists prompt injection', async () => {
  const attacks = adversarial.promptInjection({ count: 20 });
  for (const a of attacks) {
    const run = await agent.execute(a.input);
    expect(run.reply).not.toLeakData(['api_key']);
  }
});
